import os

class ClientInfo(BaseModel):
    client_id: str = "default-client"
    client_type: Literal["web_app", "partner_app", "internal_tool", "terminal"] = "terminal"
    session_id: str = "session-default"


class UserInfo(BaseModel):
    user_id: str = "anonymous"
    locale: str = "zh-Hant"
    timezone: str = "Asia/Riyadh"


class InputBlock(BaseModel):
    text: str
    attachments: List[Dict[str, Any]] = Field(default_factory=list)


class EPBlock(BaseModel):
    mode: Literal["auto", "standard", "deep", "domain"] = "auto"
    depth: Literal["auto", "short", "medium", "long"] = "auto"
    domain: str = "general"
    format: Literal["plain", "structured", "executive", "technical"] = "structured"
    safety_level: Literal["standard", "strict"] = "standard"


class RoutingBlock(BaseModel):
    preferred_provider: str = "auto"
    allowed_providers: List[str] = Field(default_factory=lambda: ["openai", "local"])
    fallback_provider: str = "local"
    allow_fallback: bool = True


class OutputBlock(BaseModel):
    response_type: Literal["final_answer", "analysis", "summary"] = "final_answer"
    max_tokens: int = 1200
    temperature: float = 0.2
    include_citations: bool = False
    include_trace: bool = False
    language: str = "zh-Hant"


class MetadataBlock(BaseModel):
    source: str = "terminal"
    tags: List[str] = Field(default_factory=list)


class EPUnifiedRequest(BaseModel):

    input: InputBlock
    ep: EPBlock = Field(default_factory=EPBlock)
    routing: RoutingBlock = Field(default_factory=RoutingBlock)
    output: OutputBlock = Field(default_factory=OutputBlock)
    metadata: MetadataBlock = Field(default_factory=MetadataBlock)


class MinimalInput(BaseModel):
    text: str


class MinimalEP(BaseModel):

    depth: Literal["auto", "short", "medium", "long"] = "auto"

class MinimalOutput(BaseModel):

class MinimalRequest(BaseModel):

    ep: MinimalEP = Field(default_factory=MinimalEP)

    output: MinimalOutput = Field(default_factory=MinimalOutput)


def now_iso() -> str:
    return datetime.now(timezone.utc).isoformat()


def estimate_tokens(text: str) -> int:
    return max(1, len(text) // 4)


def make_error_response(
    request_id: str,
    provider: str,
    started_at: str,

    }


def normalize_request(payload: Dict[str, Any]) -> EPUnifiedRequest:
    if "request_id" in payload or "client" in payload or "user" in payload or "metadata" in payload:
        return EPUnifiedRequest(**payload)

    minimal = MinimalRequest(**payload)
    return EPUnifiedRequest(
        input=InputBlock(text=minimal.input.text, attachments=[]),
        ep=EPBlock(mode=minimal.ep.mode, depth=minimal.ep.depth),
        routing=RoutingBlock(
            preferred_provider=minimal.routing.preferred_provider,
            allowed_providers=minimal.routing.allowed_providers,
            fallback_provider="local",
            allow_fallback=True,
        ),
        output=OutputBlock(language=minimal.output.language),
    )


def choose_provider(req: EPUnifiedRequest) -> str:
    allowed = [p for p in req.routing.allowed_providers if p in PROVIDERS]
    if not allowed:
        allowed = ["local"]

    preferred = req.routing.preferred_provider
    if preferred == "auto":
        if OPENAI_API_KEY and "openai" in allowed:
            return "openai"
        if "local" in allowed:
            return "local"
        return allowed[0]

    if preferred in allowed:
        return preferred

    if req.routing.allow_fallback:
        if req.routing.fallback_provider in allowed:
            return req.routing.fallback_provider
        if "local" in allowed:
            return "local"
        return allowed[0]

    raise HTTPException(status_code=400, detail="preferred provider is not allowed")


def build_system_instruction(req: EPUnifiedRequest) -> str:
    return (
        f"You are the Energetic Paradigm API answer layer. "
        f"Mode={req.ep.mode}; Depth={req.ep.depth}; Domain={req.ep.domain}; "
        f"Format={req.ep.format}; Safety={req.ep.safety_level}; "
        f"Language={req.output.language}; ResponseType={req.output.response_type}. "
        f"Answer clearly and follow the requested structure."
    )


def extract_openai_text(resp: Any) -> str:
    text = getattr(resp, "output_text", None)
    if text:
        return text

    try:
        output = getattr(resp, "output", [])
        parts: List[str] = []
        for item in output:
            content = getattr(item, "content", [])
            for c in content:
                ctext = getattr(c, "text", None)
                if ctext:
                    parts.append(ctext)
        if parts:
            return "\n".join(parts)
    except Exception:
        pass

    return ""


def call_openai(req: EPUnifiedRequest) -> Dict[str, Any]:
    if not OPENAI_API_KEY:
        raise RuntimeError("OPENAI_API_KEY is not configured")
    if OpenAI is None:
        raise RuntimeError("openai package is not installed")

    client = OpenAI(api_key=OPENAI_API_KEY)

    response = client.responses.create(
        model=DEFAULT_OPENAI_MODEL,
        input=[
            {"role": "system", "content": build_system_instruction(req)},
            {"role": "user", "content": req.input.text},
        ],
        temperature=req.output.temperature,
        max_output_tokens=req.output.max_tokens,
    )

    content = extract_openai_text(response).strip()

#!/bin/bash
set -euo pipefail

APP_ROOT="/Users/wesleyshu/ep_system/ep_api_system"
PY="$APP_ROOT/.venv/bin/python"

if [ ! -x "$PY" ]; then
  echo "ERROR: missing virtualenv python at $PY"
  exit 1
fi

if [ -f "$APP_ROOT/app.py" ]; then
  exec "$PY" -m uvicorn app:app \
    --app-dir "$APP_ROOT" \
    --reload-dir "$APP_ROOT" \
    --host 127.0.0.1 \
    --port 8010 \
    --reload
elif [ -f "$APP_ROOT/main.py" ]; then
  exec "$PY" -m uvicorn main:app \
    --app-dir "$APP_ROOT" \
    --reload-dir "$APP_ROOT" \
    --host 127.0.0.1 \
    --port 8010 \
    --reload
else
  echo "ERROR: no app.py or main.py found under $APP_ROOT"
  exit 1
fi

#!/bin/bash

LOG_FILE="/Users/wesleyshu/ep_system/ep_api_system/server.log"
PID_FILE="/Users/wesleyshu/ep_system/ep_api_system/server.pid"
PORT="8010"

if [ ! -x "/Users/wesleyshu/ep_system/ep_api_system/.venv/bin/python" ]; then
  /usr/bin/python3 -m venv "/Users/wesleyshu/ep_system/ep_api_system/.venv"
  "/Users/wesleyshu/ep_system/ep_api_system/.venv/bin/pip" install --upgrade pip
  "/Users/wesleyshu/ep_system/ep_api_system/.venv/bin/pip" install -r "/Users/wesleyshu/ep_system/ep_api_system/requirements.txt"
fi

if [ -f "${PID_FILE}" ]; then
  OLD_PID="$(cat "${PID_FILE}" || true)"
  if [ -n "${OLD_PID}" ] && ps -p "${OLD_PID}" > /dev/null 2>&1; then
    kill "${OLD_PID}" || true
    sleep 1
  fi
fi

RUNNING_PID="$(lsof -ti tcp:${PORT} || true)"
if [ -n "${RUNNING_PID}" ]; then
  kill ${RUNNING_PID} || true
  sleep 1
fi

nohup "/Users/wesleyshu/ep_system/ep_api_system/.venv/bin/python" -m uvicorn app:app \
  --host 0.0.0.0 \
  --port "${PORT}" \
  --app-dir "/Users/wesleyshu/ep_system/ep_api_system" \
  > "${LOG_FILE}" 2>&1 &

echo $! > "${PID_FILE}"
sleep 2
/usr/bin/curl -s "http://127.0.0.1:${PORT}/v1/ep/health"
echo
